## Epsilon
```py
class Solution:
    def maxFrequencyElements(self, nums: List[int]) -> int:
        d = defaultdict(int)

        for i in list(set(nums)):
            d[nums.count(i)] += 1
        
        max_key = max(d.keys())
        return max_key*d[max_key]
```

### Hojun
```kotlin
class Solution {
    fun maxFrequencyElements(nums: IntArray): Int {
        var ans = 0
        var max = 0
        for(num in nums.distinct()){
            var cnt = nums.count{it -> it == num}
            if(max < cnt){
                ans = 1
                max = cnt
            }else if(max == cnt){
                ans++
            }
        }
        return ans*max
    }
}
```
